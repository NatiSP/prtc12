def []=(filas_dis, columnas_dis, valor)
  if valor != cero
    @matrix_d[AccHash.new(filas_dis, columnas_dis)] = valor
  else
    @matrix_d.delete(AccHash.new(filas_dis, columnas_dis))
  end
end
